function doGet(e) {
  var page = e && e.parameter && e.parameter.page ? e.parameter.page : '';

  // Default to 'Index' if no page is specified
  if (!page) {
    return HtmlService.createTemplateFromFile('Index')
      .evaluate()
      .setTitle('WORLDFORMERPBOOKS - Home')
      .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
  }

  // Handle specific pages
  try {
    return HtmlService.createTemplateFromFile(page)
      .evaluate()
      .setTitle('WORLDFORMERPBOOKS - ' + page)
      .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
  } catch (err) {
    // Fallback error page
    return HtmlService.createHtmlOutput(
      '<h2>Error: Page not found</h2><p>' + err.message + '</p><a href="' + getScriptUrl() + '">Go Home</a>'
    );
  }
}

// Helper function to include other HTML files (useful for headers/footers if needed)
function include(filename) {
  return HtmlService.createHtmlOutputFromFile(filename).getContent();
}

// Returns the URL of the deployed script
function getScriptUrl() {
  return ScriptApp.getService().getUrl();
}
